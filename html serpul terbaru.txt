  <script>
    (function () {
      function cloneMenuItems($sourceRow) {
        var $items = $sourceRow.find('a.col').clone(true, true);

        $items = $items.filter(function () {
          var label = $(this).find('p').first().text().trim().toLowerCase();
          return label !== 'menu lain' && label !== 'lainya' && label !== 'lainnya';
        });

        $items = $items.filter(function () {
          var href = ($(this).attr('href') || '').trim();
          var label = $(this).find('p').first().text().trim().toLowerCase();
          return href !== 'MOBILE_PRABAYAR/24' && label !== 'uang digital bebas nominal';
        });

        $items = $items.filter(function () {
          var href = ($(this).attr('href') || '').toLowerCase().trim();
          return href !== 'javascript:void(0)';
        });

        $items.removeClass('hide-main').css('display', '');
        return $items;
      }

      function sectionHtml(title) {
        return (
          '<div class="pt-1 pb-2">' +
          '<strong>' + title + '</strong>' +
          '<hr class="my-2"/>' +
          '</div>' +
          '<div class="row row-cols-4 px-1 menu-section"></div>'
        );
      }

      function renderAllMenus() {
        var $container = $('#allMenuContainer');
        $container.empty();

        $container.append(sectionHtml('Prabayar'));
        $container.append('<div class="pb-3"></div>');
        $container.append(sectionHtml('Pascabayar'));

        var $sections = $container.find('.menu-section');
        var $prabayarSection = $sections.eq(0);
        var $pascabayarSection = $sections.eq(1);

        $prabayarSection.append(cloneMenuItems($('#menuPrabayar')));
        $pascabayarSection.append(cloneMenuItems($('#menuPascabayar')));

        /* âœ… TAMBAH MENU KHUSUS MODAL */
        $prabayarSection.append(`
          <a class="col p-1 text-decoration-none text-reset" href="MOBILE_PRABAYAR/25">
            <div class="d-flex flex-column align-items-center text-center p-0">
              <div class="card border-0 shadow-sm rounded-3 p-2 d-flex justify-content-center align-items-center card-old-android"
                   style="height:50px;width:50px">
                <img height="45" width="45"
                     src="https://sp2680.serpul.co.id/img/icon/telkomsel-omni-20260120153343.png"
                     alt="Telkomsel Omni" />
              </div>
              <p class="mt-2 m-0 p-0" style="font-size:small">Telkomsel Omni</p>
            </div>
          </a>
        `);
      }

      $('#allMenuModal').on('show.bs.modal', function () {
        renderAllMenus();
        $('#menuSearch').val('');
        $('#allMenuContainer a.col').show();
      });

      $('#menuSearch').on('input', function () {
        var q = ($(this).val() || '').toLowerCase().trim();
        $('#allMenuContainer a.col').each(function () {
          var text = $(this).text().toLowerCase();
          $(this).toggle(text.indexOf(q) !== -1);
        });
      });
    })();
  </script>
